```typescript
class UserProfile {
    userName: string

    constructor(dto: UserProfileDTO) {
        this.userName = dto.userName
    }
}
```

```typescript
class User {
    id: UserID
    profile: UserProfile

    constructor(dto: UserDTO) {
        this.id = dto.id
        this.profile = new UserProfile(dto.profile)
    }
    
    // In newer versions of angular you can use inject(UserRepository) instead
    updateProfile(repo: UserRepositoryInterface, profile: UserProfile) {
        return repo.update(profile).pipe(map(res => {
            this.profile = profile
            return this
        }))
    }
}
```
